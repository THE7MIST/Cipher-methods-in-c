   [[ $num =~ [A-Z] ]] && \
   [[ $num =~ [a-z] ]] && \
   [[ $num =~ [#!@\$%\^&*()_+=-] ]] && \
   [[ ${#num} -ge 8 && ${#num} -le 20 ]]

    echo "Strong Password"
else
